root		:			statements
			|

statements	:			statements statement
			|			statements ; statement
			|			statement

statement	:			expr
			|			CALC ID
			|			EXIT
			|			VARS
			|			NEW
			|			POLAR expr

expr		:			expr + expr
			|			expr - expr
			|			expr * expr
			|			expr / expr
			|			expr ^ expr
			|			- expr						%prec UMINUS
			|			mini_term
			|			NUMBER
			|			ID ASSIGN expr
			|			expr THEN expr
			|			expr THEN ' ID expr '

mini_term	:			ID
			|			NUMBER mini_term			%prec CONSTANT
			|			NUMBER mini_term POW expr
			|			MATHFUNC expr
			|			LPAREN expr RPAREN

------

precedence = (
    ('right', ASSIGN),
    ('left', THEN),
    ('left', PLUS, MINUS),
    ('left', TIMES, DIV),
    ('right', CONSTANT),
    ('right', MATHFUNC, UMINUS),
    ('right', POW),
)
