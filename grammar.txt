root		:			statements

statements	:			statements statement
			|			statement

statement	:			expr
			|			ID = expr
			|			CALC ID
			|			EXIT

expr		:			expr + expr
			|			expr - expr
			|			expr * expr
			|			expr / expr
			|			expr ^ expr
			|			- expr
			|			MATHFUNC expr
			|			( expr )
			|			NUMBER
			|			ID
			|			expr THEN formula
