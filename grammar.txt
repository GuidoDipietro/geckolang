root		:			statements
			|

statements	:			statements statement
			|			statements ; statement
			|			statement

statement	:			expr
			|			CALC ID
			|			EXIT
			|			VARS
			|			NEW
			|			POLAR expr

expr		:			expr + expr
			|			expr - expr
			|			expr * expr
			|			expr / expr
			|			expr ^ expr
			|			- expr
			|			MATHFUNC expr
			|			NUMBER ID
			|			( expr )
			|			NUMBER
			|			IMNUMBER
			|			ID
			|			ID ASSIGN expr
			|			expr THEN expr
			|			expr THEN ' ID expr '
